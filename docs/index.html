setdoc
Introduction

This project helps to set the doc string.

Installation
Implementation
import dataclasses
from typing import *

__all__ = ["setdoc"]


@dataclasses.dataclass
class setdoc:
    'A class to set doc strings.'
    doc: Any

    def __call__(self, target: Any) -> Any:
        'Set the doc string of the passed target ' \
        'to the value stored in the doc field ' \
        'of the setdoc object.'
        target.__doc__ = self.doc
        return target
Usage
from setdoc import setdoc

# Create a setdoc instance with the desired docstring
doc_updater = setdoc("This function adds two numbers.")

# Apply it to a target function
@doc_updater
def add(a, b):
    return a + b

# The function now has an updated docstring
print(add.__doc__)
# Output: This function adds two numbers.
